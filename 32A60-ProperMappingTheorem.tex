\begin{thm}
Let $X$ and $Y$ be complex manifolds, $f \colon X \to Y$ be a \PMlinkname{proper}{ProperMap} holomorphic map,
and $V \subset X$ be a complex analytic subvariety of $X.$
Then $f(V)$ is a subvariety of $Y.$
\end{thm}
